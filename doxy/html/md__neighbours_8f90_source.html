56 subroutine find_neighbours(nl,atoms,group1,group2,box)
57  type(particles):: atoms
58  type(particle_group):: group1,group2
59  type(simulation_cell):: box
60  type(neighbour_list):: nl
61  real:: dr(3),dr2
62  integer:: i,j,k,ind,jnd,nnumind,lessnnumind
63 
64  if (group1%N>nl%N) then; write(*,*) 'error: group1%N>nl%N',group1%N,nl%N; stop; endif
65  !$OMP PARALLEL firstprivate(i,j,k,ind,jnd,dr,dr2,nnumind,lessnnumind)
66  !$OMP DO
67  do ind=1,group1%N
68  i = group1%indexes(ind)
69  nl%particle_index(ind) = group1%indexes(ind)
70  !nl%nlist(:nl%nnum(ind),ind) = 0
71  nnumind = 0
72  lessnnumind = -1
73  do jnd=1,group2%N
74  j = group2%indexes(jnd)
75  if (i/=j) then
76  call find_distance(dr,dr2,atoms%positions(:,i),atoms%positions(:,j),box)
77  if (dr2<nl%r_cut*nl%r_cut) then
78  if (lessnnumind==-1 .and. i<j) lessnnumind = nnumind
79  nnumind = nnumind+1
80  if (nnumind>nl%neighb_num_max) then; write(*,*) 'error: too many neighbours',ind,nnumind; stop; endif
81  nl%nlist(nnumind,ind) = jnd
82  nl%dr(1,nnumind,ind) = dr(1)
83  nl%dr(2,nnumind,ind) = dr(2)
84  nl%dr(3,nnumind,ind) = dr(3)
85  nl%moddr(nnumind,ind) = sqrt(dr2)
86  endif
87  endif
88  enddo
89  if(lessnnumind==-1) then
90  nl%lessnnum(ind) = nnumind
91  else
92  nl%lessnnum(ind) = lessnnumind
93  endif
94  nl%nnum(ind) = nnumind
95  enddo
96  !$OMP END DO
97  !$OMP END PARALLEL
98 
99  return
100 end subroutine find_neighbours
